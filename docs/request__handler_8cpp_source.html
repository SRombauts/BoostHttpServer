30 void request_handler::handle_request(const request& req, reply& rep)
31 {
32  // Decode url to path.
33  std::string request_path;
34  if (!url_decode(req.uri, request_path))
35  {
36  reply::stock_reply(reply::bad_request, rep);
37  return;
38  }
39 
40  // Decode and tokenize the query part of the URI
41  std::string query;
42  options_map get_params;
43  if (!url_decode(req.query, query))
44  {
45  reply::stock_reply(reply::bad_request, rep);
46  return;
47  }
48  query_tokenize (query, get_params);
49 
50  // Decode and tokenize the content part of the POST request
51  std::string content;
52  options_map post_params;
53  if (!url_decode(req.content, content))
54  {
55  reply::stock_reply(reply::bad_request, rep);
56  return;
57  }
58  query_tokenize (content, post_params);
59 
60  // Check if the request is for a registered dynamic resource
61  resource_map::iterator resource = resource_map_.find(request_path);
62  if (resource_map_.end() != resource)
63  {
64  // A dynamic web page will be generated in reply to the client request
65  (*resource).second (req, rep);
66  return;
67  }
68 
69  // Request path must be absolute and not contain "..".
70  if (request_path.empty() || request_path[0] != '/'
71  || request_path.find("..") != std::string::npos)
72  {
73  reply::stock_reply(reply::bad_request, rep);
74  return;
75  }
76 
77  // If path ends in slash (i.e. is a directory) then add "index.html".
78  if (request_path[request_path.size() - 1] == '/')
79  {
80  request_path += "index.html";
81  }
82  else
83  {
84  // If it does not end in a slash, but is a directory, then redirect to the slash terminated path
85  if (boost::filesystem::is_directory(doc_root_ + request_path))
86  {
87  reply::redirect_reply(request_path + "/", rep);
88  return;
89  }
90  }
91 
92  // Determine the file extension.
93  std::size_t last_slash_pos = request_path.find_last_of("/");
94  std::size_t last_dot_pos = request_path.find_last_of(".");
95  std::string extension;
96  if (last_dot_pos != std::string::npos && last_dot_pos > last_slash_pos)
97  {
98  extension = request_path.substr(last_dot_pos + 1);
99  }
100 
101  // Open the file to send back.
102  std::string full_path = doc_root_ + request_path;
103  std::ifstream is(full_path.c_str(), std::ios::in | std::ios::binary);
104  if (!is)
105  {
106  reply::stock_reply(reply::not_found, rep);
107  return;
108  }
109 
110  // Fill out the reply to be sent to the client.
111  rep.status = reply::ok;
112  char buf[512];
113  while (is.read(buf, sizeof(buf)).gcount() > 0)
114  rep.content.append(buf, (unsigned int)is.gcount());
115  rep.headers.resize(2);
116  rep.headers[0].name = "Content-Length";
117  rep.headers[0].value = boost::lexical_cast<std::string>(rep.content.size());
118  rep.headers[1].name = "Content-Type";
119  rep.headers[1].value = mime_types::extension_to_type(extension);
120 }

122 bool request_handler::url_decode(const std::string& in, std::string& out)
123 {
124  out.clear();
125  out.reserve(in.size());
126  for (std::size_t i = 0; i < in.size(); ++i)
127  {
128  if (in[i] == '%')
129  {
130  if (i + 3 <= in.size())
131  {
132  int value = 0;
133  std::istringstream is(in.substr(i + 1, 2));
134  if (is >> std::hex >> value)
135  {
136  out += static_cast<char>(value);
137  i += 2;
138  }
139  else
140  {
141  return false;
142  }
143  }
144  else
145  {
146  return false;
147  }
148  }
149  else if (in[i] == '+')
150  {
151  out += ' ';
152  }
153  else
154  {
155  out += in[i];
156  }
157  }
158  return true;
159 }

161 void request_handler::query_tokenize(const std::string& in, options_map& out)
162 {
163  bool in_option_name = true;
164  std::string option_name;
165  std::string option_value;
166 
167  out.clear();
168 
169  for (std::size_t i = 0; i < in.size(); ++i)
170  {
171  if (in_option_name)
172  {
173  // parsing the name of an option
174  if (in[i] == '=')
175  {
176  in_option_name = false;
177  }
178  else if (in[i] == '&')
179  {
180  out[option_name] = ""; // option without value
181  option_name.clear ();
182  }
183  else
184  {
185  option_name += in[i];
186  }
187  }
188  else
189  {
190  // parsing the value of an option
191  if (in[i] == '&')
192  {
193  out[option_name] = option_value; // option with value
194  option_name.clear ();
195  option_value.clear ();
196  in_option_name = true;
197  }
198  else
199  {
200  option_value += in[i];
201  }
202  }
203  }
204 
205  if (false == option_name.empty())
206  {
207  out[option_name] = option_value; // last option with or without value
208  }
209 }
