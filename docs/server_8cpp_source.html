18 server::server(const std::string& address, const std::string& port,
19  const std::string& doc_root)
20  : io_service_(),
21  acceptor_(io_service_),
22  connection_manager_(),
23  new_connection_(new connection(io_service_,
24  connection_manager_, request_handler_)),
25  request_handler_(doc_root)
26 {
27  // Open the acceptor with the option to reuse the address (i.e. SO_REUSEADDR).
28  boost::asio::ip::tcp::resolver resolver(io_service_);
29  boost::asio::ip::tcp::resolver::query query(address, port);
30  boost::asio::ip::tcp::endpoint endpoint = *resolver.resolve(query);
31  acceptor_.open(endpoint.protocol());
32  acceptor_.set_option(boost::asio::ip::tcp::acceptor::reuse_address(true));
33  acceptor_.bind(endpoint);
34  acceptor_.listen();
35  acceptor_.async_accept(new_connection_->socket(),
36  boost::bind(&server::handle_accept, this,
37  boost::asio::placeholders::error));
38 }

40 void server::run()
41 {
42  // The io_service::run() call will block until all asynchronous operations
43  // have finished. While the server is running, there is always at least one
44  // asynchronous operation outstanding: the asynchronous accept call waiting
45  // for new incoming connections.

56 void server::handle_accept(const boost::system::error_code& e)
57 {
58  if (!e)
59  {
60  connection_manager_.start(new_connection_);
61  new_connection_.reset(new connection(io_service_,
62  connection_manager_, request_handler_));
63  acceptor_.async_accept(new_connection_->socket(),
64  boost::bind(&server::handle_accept, this,
65  boost::asio::placeholders::error));

69 void server::handle_stop()
