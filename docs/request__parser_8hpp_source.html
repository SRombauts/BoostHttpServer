33  /// Parse some data. The tribool return value is true when a complete request
34  /// has been parsed, false if the data is invalid, indeterminate when more
35  /// data is required. The InputIterator return value indicates how much of the
36  /// input has been consumed.
37  template <typename InputIterator>
38  boost::tuple<boost::tribool, InputIterator> parse(request& req,
39  InputIterator begin, InputIterator end)
40  {
41  while (begin != end)
42  {
43  boost::tribool result = consume(req, *begin++);
44  if (result || !result)
45  return boost::make_tuple(result, begin);
46  }
47  boost::tribool result = boost::indeterminate;
48  return boost::make_tuple(result, begin);
49  }
