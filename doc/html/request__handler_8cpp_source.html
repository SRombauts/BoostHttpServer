00033   std::string request_path;

00064     // A dynamic web page will be generated in reply to the client request
00065     (*resource).second (req, rep);
00066     return;
00067   }
00068 
00069   // Request path must be absolute and not contain "..".
00070   if (request_path.empty() || request_path[0] != '/'
00071       || request_path.find("..") != std::string::npos)
00072   {
00073     reply::stock_reply(reply::bad_request, rep);
00074     return;
00075   }
00076

00080     request_path += "index.html";

00082   else

00085     if (boost::filesystem::is_directory(doc_root_ + request_path))
00086     {
00087       reply::redirect_reply(request_path + "/", rep);

00089     }
00090   }
00091

00093   std::size_t last_slash_pos = request_path.find_last_of("/");

00162 {
00163   bool        in_option_name = true;
00164   std::string option_name;
00165   std::string option_value;

00170   {
00171     if (in_option_name)
00172     {
00173       // parsing the name of an option

00183       else
00184       {
00185         option_name += in[i];

00187     }
00188     else
00189     {
00190       // parsing the value of an option
00191       if (in[i] == '&')
00192       {
00193         out[option_name] = option_value; // option with value
00194         option_name.clear ();
00195         option_value.clear ();
00196         in_option_name = true;
00197       }
00198       else
00199       {
00200         option_value += in[i];
00201       }
00202     }
00203   }
00204 
00205   if (false == option_name.empty())
00206   {
00207     out[option_name] = option_value; // last option with or without value
