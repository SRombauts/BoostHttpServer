00105 std::vector<boost::asio::const_buffer> reply::to_buffers()
00106 {
00107   std::vector<boost::asio::const_buffer> buffers;
00108   
00109   buffers.push_back(status_strings::to_buffer(status));
00110   for (std::size_t i = 0; i < headers.size(); ++i)
00111   {
00112     const header& h = headers[i];
00113     buffers.push_back(boost::asio::buffer(h.name));
00114     buffers.push_back(boost::asio::buffer(misc_strings::name_value_separator));
00115     buffers.push_back(boost::asio::buffer(h.value));
00116     buffers.push_back(boost::asio::buffer(misc_strings::crlf));
00117   }
00118   buffers.push_back(boost::asio::buffer(misc_strings::crlf));
00119   buffers.push_back(boost::asio::buffer(content));
00120   return buffers;
